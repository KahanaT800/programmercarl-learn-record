```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;
int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

void bfs(vector<vector<int>>& grid, int x, int y) {
    if (grid[x][y] != 1) return;
    queue<pair<int, int>> q;
    q.push({x, y});
    grid[x][y] = 2;

    while (!q.empty()) {
        auto cur = q.front();
        q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = cur.first + dir[i][0];
            int ny = cur.second + dir[i][1];
            if (nx < 0 || nx >= grid.size() || ny < 0 || ny >= grid[0].size()) continue;
            if (grid[nx][ny] != 1) continue;
            grid[nx][ny] = 2;
            q.push({nx, ny});
        }
    }
}

int main() {
    int n, m, s, t;
    cin >> n >> m;
    vector<vector<int>> grid(n, vector<int>(m, 0));

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    for (int i = 0; i < n; i++) {
        bfs(grid, i, 0);
        bfs(grid, i, m - 1);
    }

    for (int j = 0; j < m; j++) {
        bfs(grid, 0, j);
        bfs(grid, n - 1, j);
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                grid[i][j] = 0;
            } else if (grid[i][j] == 2) {
                grid[i][j] = 1;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m - 1; j++) {
            cout << grid[i][j] << " ";
        }
        cout << grid[i][m - 1] << endl;
    }
}
```
