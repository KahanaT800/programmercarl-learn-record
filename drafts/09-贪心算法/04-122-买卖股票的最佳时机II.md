### 代码

```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int profit = 0;
        for (int i = 1; i < prices.size(); i++) {
            // 只要后一天的股价高于前一天的股价就可以买进卖出，这样一定获利
            // 多次交易只是把一段长的收入拆分成了几段小的
            if (prices[i] > prices[i - 1]) {
                profit += prices[i] - prices[i-1];
            }
        }
        return profit;
    }
};
```
