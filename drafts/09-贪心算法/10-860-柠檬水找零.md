### 代码

```cpp
class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int five = 0;
        int ten = 0;
        for (int b : bills) {
            if (b == 5) {
                five++;
            }
            if (b == 10) {
                if (five < 1) {
                    return false;
                } else {
                    five--;
                    ten++;
                }
            }
            if (b == 20) {
                // 至少需要一张5元， 而且如果 10元少于 1张就需要3张5元
                if (five < 1 || (five < 3 && ten < 1)) {
                    return false;
                } else {
                    // 至少有一张10元就先用10元
                    if (ten >= 1) {
                        ten--;
                        five--;
                    } else {
                        five = five - 3;
                    }
                }
            }
        }

        return true;
    }
};
```
