### 代码

```cpp
class Solution {
public:
    int countSubstrings(string s) {
        int n = s.size();
        // dp[i][j] 表示从下标 i 到下标 j 的子串是不是回文
        vector<vector<int>> dp(n, vector<int>(n, false));
        int sum = 0;

        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j < n; j++) {
                // 如果 s[i] != s[j]，两端不一样，肯定不是回文
                // 如果 s[i] == s[j]，两端一样，那要看里面的部分
                if (s[i] == s[j]) {
                    if (j - i <= 1) {
                        // j - 1 <= 1, 没有字符和1个字符都是回文串
                        dp[i][j] = true;
                    } else {
                        // 否则看左右往内移动一步的是不是回文串
                        // 这里的条件由 j - 1 <= 1 保证不超限
                        dp[i][j] = dp[i+1][j-1];
                    }
                }
                if (dp[i][j] == true) {
                    sum++;
                }
            }
        }
        return sum;
    }
};
```
