```cpp
class Solution {
public:
    int maxProfit(int k, vector<int>& prices) {
        int n = prices.size();
        vector<vector<int>> dp(n, vector<int>( 2 * k + 1, 0));
        // 2 * k + 1种状态
        // 0 ：不买不卖
        // 2 * k + 1 ：第K次持有股票
        // 2 * k ：第K次卖出股票
        for (int j = 1; j < 2 * k + 1; j++) {
            if (j % 2 == 1) {
                dp[0][j] = -prices[0];
            }
        }
        for (int i = 1; i < n; i++) {
            dp[i][0] = dp[i - 1][0];
            for (int j = 1; j < 2 * k + 1; j++) {
                if (j % 2 == 1) {
                    dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - 1] - prices[i]);
                } else {
                    dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - 1] + prices[i]);
                }
            }
        }
        return dp[n - 1][2 * k]; 
    }
};
```
