### 代码

```cpp
class Solution {
public:
    int findLengthOfLCIS(vector<int>& nums) {
        int n = nums.size();
        int prev = 1;
        int res = 1;

        for (int i = 1; i < n; i++) {
            if (nums[i - 1] < nums[i]) {
                prev += 1;
            } else {
                prev = 1;
            }
            res = max(res, prev);
        }

        return res;
    }
};
```
