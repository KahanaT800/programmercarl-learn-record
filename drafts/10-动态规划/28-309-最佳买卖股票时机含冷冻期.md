### 实现中遇到的困难
注意最后返回要考虑所有不持有股票的情况；

### 代码

```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        vector<vector<int>> dp(n, vector<int>(4, 0));
        // 4种状态
        // 0 ：还没买过，或冷静期结束后的空闲
        // 1 ：持有股票
        // 2 ：卖出股票
        // 3 ：冷静期
        dp[0][1] = -prices[0];
        for (int i = 1; i < n; i++) {
            // 第 i 天啥也不干要么是之前也没干过，要么是之前是冷近期
            dp[i][0] = max(dp[i - 1][0], dp[i - 1][3]);
            // 第 i 天持有股票要么是前一天已经持有，要么是前一天啥也没做之后买入
            dp[i][1] = max(dp[i - 1][1], max(dp[i - 1][0], dp[i - 1][3]) - prices[i]);
            // 第 i 天卖出股票是前一天是持有状态这天卖出
            dp[i][2] = dp[i - 1][1] + prices[i];
            // 第 i 天冷静期就是前一天卖出
            dp[i][3] = dp[i - 1][2];
        }
        // 需要返回所有不持有股票的状态
        return max({dp[n - 1][0], dp[n - 1][2], dp[n - 1][3]});
    }
};
```
