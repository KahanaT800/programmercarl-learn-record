### 代码

```cpp
#include <iostream>
#include <string>
using namespace std;

string addNumber(string& s) {
    string result;
    for (char c : s) {
        if (c >= '0' && c <= '9') {
            result += "number";
        } else {
            result += c;
        }
    }
    return result;
}

int main() {
    string s;
    getline(cin, s);

    cout << addNumber(s) << endl;

    return 0;
}
```
