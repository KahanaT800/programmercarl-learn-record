```cpp
class Solution {
public:
    string reverseStr(string s, int k) {
        int n = s.size();
        int time = n / (2* k);
        int tail = n % (2* k);
        int index = 0;

        while (time > 0) {
            swapStr(s, index, index+k-1);
            index = index + 2 * k ;
            time--;
        }

        if (tail > 0 && tail < k) {
            swapStr(s, index, index+tail-1);
        } else {
            swapStr(s, index, index+k-1);
        }

        return s;
    }

    void swapStr(string& s, int l, int r) {
        int left = l, right = r;

        while (left < right) {
            char tmp = s[left];
            s[left] = s[right];
            s[right] = tmp;
            left++;
            right--;
        }
    }
};
```
