### 第一想法
没有理解如何跑两圈

### 看完题解后的想法
2倍长度，使用 % 进行重定位

### 实现中遇到的困难
注意压的是序号

### 代码

```cpp
class Solution {
public:
    vector<int> nextGreaterElements(vector<int>& nums) {
        int n = nums.size();
        stack<int> st;
        vector<int> res(n, -1);
        // 跑两圈
        for (int i = 0; i < 2 * n; i++) {
            while (!st.empty() && nums[st.top()] < nums[i % n]) {
                res[st.top()] = nums[i % n];
                st.pop();
            }
            // 只有第一圈才记录
            if (i < n) st.push(i);
        }

        return res;
    }
};
```
