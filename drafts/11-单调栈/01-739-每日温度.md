### 第一想法
维护一个元素递增的单调栈存放数组下标，这样比较top元素和新元素，如果新元素大于top则说明至少top指向的位置找到了更高的，需要记录并弹出；

### 代码

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> ans(n, 0);
        stack<int> st;
        for (int i = 0; i < n; i++) {
            while (!st.empty() && temperatures[i] > temperatures[st.top()]) {
                int top = st.top();
                st.pop();
                ans[top] = i - top; 
            }
            st.push(i);
        }

        return ans;
    }
};
```
