### 实现中遇到的困难
ACM输入输出忘了咋实现了；

注意因为是a到b的区间，所以下标应该是 `b - (a - 1)`;

### 代码

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, a, b;
    cin >> n;

    vector<int> input(n);
    vector<int> prevs(n);

    int prev = 0;

    for (int i = 0; i < n; i++) {
        cin >> input[i];
        prev += input[i];
        prevs[i] = prev;
    }

    while (cin >> a >> b) {
        int result = 0;
        if (a == 0) {
            result = prevs[b];
        } else {
            result = prevs[b] - prevs[a-1];
        }
        cout << result << endl;
    }

    return 0;
}
```
