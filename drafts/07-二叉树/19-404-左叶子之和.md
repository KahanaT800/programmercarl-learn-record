### 代码

```cpp
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        int sum = 0;
        dfs(root, sum, false);
        return sum;
    }

    void dfs(TreeNode* root, int& sum, bool isLeft) {
        if (!root) {
            return;
        }

        if (!root->left && !root->right && isLeft) {
            sum += root->val;
            return;
        }

        dfs(root->left, sum, true);
        dfs(root->right, sum, false);
    }
};
```
