### 实现中遇到的困难
需要对最左节点进行一个特判，因为最左节点没有可以相减的；

### 代码

```cpp
class Solution {
public:
    int res = INT_MAX;
    int prev = -1;  // -1 表示还没访问过任何节点

    void dfs(TreeNode* root) {
        if (!root) return;

        dfs(root->left);

        if (prev != -1) {
            res = min(res, root->val - prev);  // BST中序是升序，不需要abs
        }
        prev = root->val;

        dfs(root->right);
    }

    int getMinimumDifference(TreeNode* root) {
        dfs(root);
        return res;
    }
};
```
