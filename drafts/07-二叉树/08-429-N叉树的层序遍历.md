```cpp
class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
         vector<vector<int>> res;
        if (!root) {
            return res;
        }
        queue<Node*> q;
        q.push(root);

        while (!q.empty()) {
            vector<int> r;
            int num = q.size();
            for (int i = 0; i < num; i++) {
                Node* cur = q.front();
                q.pop();
                r.push_back(cur->val);

                int size = cur->children.size();
                for (int j = 0; j < size; j++) {
                    q.push(cur->children[j]);
                }
            }
            res.push_back(r);
        }

        return res;
    }
};
```
